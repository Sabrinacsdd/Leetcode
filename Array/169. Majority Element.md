```java
class Solution {
    public int majorityElement(int[] nums) {
        int count = 0;
        int cur = nums[0];
        for(int n : nums){
            if(count == 0){
                cur = n;
                count++;
            }
            else if(n == cur){
                count++;
            }
            else{
                count--;
            }
        }
        return cur;
    }
}

```

## Divide and conquer
```java
class Solution {
    public int majorityElement(int[] nums) {
        return majority(nums, 0, nums.length - 1);
    }
    public int majority(int[] nums, int l, int h){
        
        if(l == h){
            return nums[l];
        }
        int mid = l + (h - l) / 2;
        int left = majority(nums, l , mid);
        int right = majority(nums, mid + 1, h);
        
        if(left == right){
            return left;
        }
        int left_count = count(nums, left, l , mid);
        int right_count = count(nums, right, mid + 1, h);
        
        return left_count > right_count ? left : right;
    }
    public int count(int[] nums, int target, int l, int h){
        int count = 0;
        for(int i = l; i <= h; i++){
            if(nums[i] == target){
                count++;
            }
        }
        return count;
    }
}
```
