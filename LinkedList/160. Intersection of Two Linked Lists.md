```java
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {
    public ListNode getIntersectionNode(ListNode headA, ListNode headB) {
        HashSet<ListNode> ha = new HashSet<>();
        
        ListNode ca = headA;
        ListNode cb = headB;
        while(ca != null){
            ha.add(ca);
            ca = ca.next;
        }
        while(cb !=null){
            if(ha.contains(cb)){
                return cb;
            }
            cb = cb.next;
        }
        return null;
    }
}

```
