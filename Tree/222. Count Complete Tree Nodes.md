## Problem description
```
Given a complete binary tree, count the number of nodes.

Note:

Definition of a complete binary tree from Wikipedia:
In a complete binary tree every level, except possibly the last, is completely filled, and all nodes in the last level are as far left as possible. It can have between 1 and 2h nodes inclusive at the last level h.

Example:

Input: 
    1
   / \
  2   3
 / \  /
4  5 6

Output: 6
```

## Binary search
```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int countNodes(TreeNode root) {
        TreeNode testl = root;
        TreeNode testr = root;
        if(root == null) return 0;
        if(root.left == null && root.right == null) return 1;
        //h is the height of the tree
        int h = 1;
        int pass = 0;
        while(testl.left != null){
            testl = testl.left;
            pass += (int) Math.pow(2, h - 1);
            h++;
        }
        int hr = 1;
        while(testr.right != null){
            testr = testr.right;
            hr++;
        }
        // h== hr means is it's a full tree
        if(h == hr) return pass + (int) Math.pow(2, h - 1);
        int l = 1;
        int r = (int) Math.pow(2, h - 1);
        while(l < r){
            int m = l + (r - l) / 2;
            if(checknull(root, m, h)){
                r = m;
            }
            else{
                l = m + 1;
            }
        }
        
        int ans = pass + l - 1;
        return ans;
    }
    public boolean checknull(TreeNode root, int m, int h){
        int total = (int) Math.pow(2, h - 1);
        while(h > 1){
            // means it's to the left this step
            if(total / m >= 2) root = root.left;
            else {
                root = root.right;
                m = m -total/ 2;
            }
            total = total/ 2;
            h--;
        }
        if(root == null) return true;
        else return false;
        
    }
}

```
