```java
class Solution {
    public int minSubArrayLen(int s, int[] nums) {
        if (nums.length == 0) {
            return 0;
        }
        int l = 0;
        int r = 0;
        int sum = 0;
        int ans = Integer.MAX_VALUE;
        //check from [i, j]
        while (r < nums.length) {
            sum += nums[r];
            if (sum >= s) {
                //current is a candidate for ans
                ans = Math.min(ans, r - l + 1);
                if (ans == 1){
                    return 1;
                }
                sum -= nums[l];
                sum -= nums[r];
                ++l;
            } else {
                ++r;
            }
        }
        return ans == Integer.MAX_VALUE ? 0 : ans;
    }
}

```
