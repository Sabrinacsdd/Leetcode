## Problem description
```
The n-queens puzzle is the problem of placing n queens on an n√ón chessboard such that no two queens attack each other.



Given an integer n, return all distinct solutions to the n-queens puzzle.

Each solution contains a distinct board configuration of the n-queens' placement, where 'Q' and '.' both indicate a queen and an empty space respectively.

Example:

Input: 4
Output: [
 [".Q..",  // Solution 1
  "...Q",
  "Q...",
  "..Q."],

 ["..Q.",  // Solution 2
  "Q...",
  "...Q",
  ".Q.."]
]
Explanation: There exist two distinct solutions to the 4-queens puzzle as shown above.
```

class Solution {
    int[] rows;
    int[] cols;
    int[] front;
    int[] back;
    List<List<String>> ans;
    public List<List<String>> solveNQueens(int n) {
        int[][] board = new int[n][n];
        ans = new ArrayList<>();
        if(n == 0) {
            ans.add(new ArrayList<>());
            return ans;
        }
        rows = new int[n];
        cols = new int[n];
        //from top to down
        //front = r + c
        front = new int[2 * n - 1];
        back = new int[2 * n - 1];
        //from right to left
        //back = r - c + n - 1;
        fill(board, 0, 0, n);

        return ans;
    }
    public void addAns(int[][] board){
        List<String> layer = new ArrayList<>();
        for(int i = 0; i < board.length; i++){
            //List<String> cur = new ArrayList<>();
            String cur = new String();
            for(int j = 0; j < board.length; j++){
                if(board[i][j] == 1){
                    cur += "Q";
                }else{
                    cur +=".";
                }
                //cur += (String.valueOf(board[i][j]));
            }
            layer.add(cur);
        }
        ans.add(layer);
    }
    public void fill(int[][] board, int r, int c, int n){
        if(r >= board.length && n == 0){
            addAns(board);
            return ;
        }
        if(r >= board.length){
            return ;
        }
        int nc = (c + 1) % board.length;
        int nr = nc == 0 ? r + 1 : r;
        
        int[] tR = rows;
        int[] tC = cols;
        int[] tF = front;
        int[] tB = back;
        //check surronding conditions
        //check rows, cols, front and end, if no array has q, we can try to put a queen here
        if(rows[r] == 0 && cols[c] == 0 && front[r + c] == 0 && back[r - c + board.length - 1] == 0){
            if(n > 0){
                board[r][c] = 1;
                rows[r] = 1;
                cols[c] = 1;
                front[r + c] = 1;
                back[r - c + board.length -1] = 1;
                //try fill current to 1
                fill(board, nr, nc, n - 1); 
                board[r][c] = 0;
                rows[r] = 0;
                cols[c] = 0;
                front[r + c] = 0;
                back[r - c + board.length -1] = 0;
            }
        }
        //try not to fill current to 1 (keep current 0)
        fill(board, nr, nc, n);
    }
}
